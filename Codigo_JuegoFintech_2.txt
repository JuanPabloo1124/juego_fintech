function onGameChange() {
  // si la pantalla central reinicio la partida, el equipo vuelve al registro
  if (teamId && game && joinedGameId && game.gameId !== joinedGameId) {
    teamId = null; teamDoc = null; joinedGameId = null; sentRound = 0; lastRoundSeen = 0;
    errMsg = "El moderador inició una partida nueva. Registra tu banco otra vez.";
  }
  // al empezar una ronda nueva, cargar decision previa o arrancar en ceros
  if (teamId && teamDoc && game && game.phase === "deciding" && game.round !== lastRoundSeen) {
    const prev = teamDoc.decisions && teamDoc.decisions[game.round];
    alloc = prev ? Object.assign({}, ZERO, prev) : Object.assign({}, ZERO);
    sentRound = prev ? game.round : 0;
    lastRoundSeen = game.round;
  }
}

function onTeamsChange() {
  // la pantalla central admite equipos nuevos mientras esta en la sala de espera
  if (mode !== "host" || !game || game.phase !== "lobby") return;
  const current = game.teams || {};
  Object.keys(teamsData).forEach(id => {
    const t = teamsData[id];
    if (t && t.gameId === game.gameId && !current[id]) {
      db.ref(ROOT + "/game/teams/" + id).set({
        name: t.name,
        color: TEAM_COLORS[Object.keys(current).length % TEAM_COLORS.length],
        ind: Object.assign({}, CONFIG.START),
        cum: Object.assign({}, ZERO)
      });
      current[id] = true;
    }
  });
}

function adjAlloc(a, d) {
  const total = sumAlloc(alloc);
  if (d > 0 && total >= CONFIG.BUDGET) return;
  alloc[a] = Math.max(0, (alloc[a]||0) + d);
  render();
}
function adjOv(id, a, d) {
  const cur = overrides[id] || Object.assign({}, ZERO);
  const total = sumAlloc(cur);
  if (d > 0 && total >= CONFIG.BUDGET) return;
  cur[a] = Math.max(0, (cur[a]||0) + d);
  overrides[id] = cur;
  render();
}
function clearOv(id) { delete overrides[id]; editing = null; render(); }

  if (!confirm("¿Borrar la partida actual y todos los equipos registrados?")) return;

  refTeams.set(null).then(createGame);
}

}

  if (game.round >= CONFIG.ROUNDS) { refGame.update({ phase: "final" }); return; }
  refGame.update({ phase: "deciding", round: game.round + 1, roundStartedAt: Date.now(), outcomes: null });
}
function resolveRound() {
  const round = game.round, ev = EVENTS[round];
  const teams = game.teams || {};
  const ids = Object.keys(teams);
  const missing = ids.filter(id => {
    const t = teamsData[id];
    return !(t && t.decisions && t.decisions[round]) && !overrides[id];
  });
  if (missing.length > 0) {
    const names = missing.map(id => teams[id].name).join(", ");
    if (!confirm("Sin decisión: " + names + ". Se les asignará inversión repartida (2 en cada área). ¿Cerrar la ronda y revelar el evento?")) return;
  }
  const g = JSON.parse(JSON.stringify(game));
  g.teams = g.teams || {};
  const outcomes = {};
  ids.forEach(id => {
    const t = g.teams[id];
    t.cum = Object.assign({}, ZERO, t.cum || {});
    const td = teamsData[id];
    const sent = td && td.decisions ? td.decisions[round] : null;
    const alloc0 = overrides[id] || sent || Object.assign({}, DEFAULT_ALLOC);
    const al = Object.assign({}, ZERO, alloc0);
    const cum = {};
    AREAS.forEach(a => { cum[a.id] = (t.cum[a.id]||0) + (al[a.id]||0); });
    const prevVal = valuation(t.ind);
    const ind = Object.assign({}, t.ind);
    ind.cap = Math.round(ind.cap + ind.cli * 0.15);
    ind.cli = Math.round(ind.cli + al.mkt * 5);
    ind.rep = clamp(Math.round(ind.rep + al.tech * 2 - 4), 0, 100);
    const tier = tierOf(ev, cum);
    const d = Object.assign({}, ev.fx[tier].d);
    let shielded = false;
    if (ev.repShield && tier !== "hi" && t.ind.rep >= 65) {
      shielded = true;
      ["cli","cap","rep"].forEach(k => { if (d[k] && d[k] < 0) d[k] = Math.round(d[k]/2); });
    }
    ind.cli = Math.max(0, Math.round(ind.cli + (d.cli||0)));
    ind.cap = Math.round(ind.cap + (d.cap||0));
    ind.rep = clamp(Math.round(ind.rep + (d.rep||0)), 0, 100);
    t.ind = ind; t.cum = cum;
    outcomes[id] = { tier: tier, d: d, shielded: shielded, defaulted: !sent && !overrides[id], prevVal: prevVal, msg: ev.fx[tier].msg };
  });
  g.phase = "results"; g.outcomes = outcomes; g.eventRound = round;
  overrides = {}; editing = null;
  refGame.set(g);
}

function join() {
  errMsg = "";
  if (!game) { errMsg = "Aún no hay partida creada. Pide a los moderadores abrir la pantalla central."; render(); return; }
  const id = slug(nameVal);
  if (!id) { errMsg = "Escribe el nombre de tu neobanco."; render(); return; }
  refTeams.child(id).once("value").then(s => {
    const existing = s.val();
    if (existing && existing.gameId === game.gameId) {
      teamDoc = existing; teamId = id; joinedGameId = game.gameId; lastRoundSeen = 0;
      onGameChange(); render();
      return;
    }
    if (game.phase !== "lobby") {
      errMsg = "La partida ya empezó. Si tu equipo ya estaba registrado, escribe el nombre EXACTO para reconectarte.";
      render(); return;
    }
    const doc = { name: nameVal.trim(), gameId: game.gameId, decisions: null };
    refTeams.child(id).set(doc).then(() => {
      teamDoc = { name: doc.name, gameId: doc.gameId, decisions: {} };
      teamId = id; joinedGameId = game.gameId; lastRoundSeen = 0;
      render();
    }).catch(() => { errMsg = "No se pudo registrar. Revisa tu conexión."; render(); });
  });
}

function submitDecision() {
  const decisions = Object.assign({}, teamDoc.decisions || {});
  decisions[game.round] = Object.assign({}, alloc);
  refTeams.child(teamId).update({ decisions: decisions }).then(() => {
    teamDoc.decisions = decisions;
    sentRound = game.round;
    errMsg = "";
    render();
  }).catch(() => { errMsg = "No se pudo enviar. Revisa tu conexión e intenta de nuevo."; render(); });
}

      '<h3 class="font-black text-lg mb-1">💼 Reparte tus ' + CONFIG.BUDGET + " fichas</h3>" +
      '<p class="text-xs text-slate-400 mb-3">Marketing trae clientes y Tecnología sube tu reputación cada ronda. Las demás áreas te protegen de lo que pueda venir. Tu inversión se acumula ronda a ronda.</p>' +
      '<div class="flex flex-col gap-2 mb-4">' + steppers + "</div>" +
      '<div class="text-center font-bold mb-3 ' + (total === CONFIG.BUDGET ? "text-emerald-400" : "text-amber-400") + '">' + (CONFIG.BUDGET - total) + " fichas por asignar</div>" +
      '<button onclick="submitDecision()" ' + (total !== CONFIG.BUDGET ? "disabled" : "") + ' class="w-full bg-emerald-500 active:bg-emerald-400 disabled:opacity-40 text-slate-950 font-black rounded-xl py-4 text-lg mb-2">' +
      (sentRound === game.round ? "✓ Enviado — actualizar decisión" : "Enviar decisión de la junta") + "</button>" +
      (sentRound === game.round ? '<p class="text-center text-xs text-emerald-400">Tu decisión llegó a la pantalla central. Puedes cambiarla hasta que cierre la ronda.</p>' : "") +
      (errMsg ? '<p class="text-center text-xs text-red-400 mt-2">' + esc(errMsg) + "</p>" : ""),
      me.name, me.color);
  }

  if (game.phase === "results") {
    const ev = EVENTS[game.eventRound];
    const o = (game.outcomes||{})[teamId];
    const ranking = Object.entries(game.teams || {}).sort((a,b) => valuation(b[1].ind) - valuation(a[1].ind));
    const myPos = ranking.findIndex(([id]) => id === teamId) + 1;
    return teamShell(
      '<div class="rounded-2xl border-2 border-red-500 bg-red-950 p-4 mb-4">' +
      '<div class="text-red-400 font-black text-xs tracking-widest mb-1">🚨 ÚLTIMA HORA</div>' +
      '<h3 class="text-xl font-black text-red-100">' + ev.title + "</h3>" +
      '<p class="text-red-200 text-sm mt-1">' + ev.desc + "</p></div>" +
      (o ? '<div class="rounded-2xl border border-slate-700 bg-slate-900 p-4 mb-4">' +
        '<span class="text-xs font-black px-3 py-1 rounded-full ' + TIER_STYLE[o.tier] + '">' + TIER_LABEL[o.tier] + "</span>" +
        '<p class="text-slate-200 text-sm mt-3">' + o.msg + (o.shielded ? " Su buena reputación amortiguó la mitad del golpe. 🛡️" : "") + "</p>" +
        '<div class="flex flex-wrap gap-2 mt-3">' + fmtDeltas(o.d) + "</div></div>" : "") +
      indicatorsHtml(me.ind) +
      '<div class="text-center mt-5 mb-3"><div class="text-4xl font-black">#' + myPos + '</div>' +
      '<div class="text-slate-400 text-sm">tu posición en el ranking</div></div>' +
      '<div class="flex flex-col gap-1">' +
      ranking.map(([id,t], i) =>
        '<div class="flex justify-between text-sm px-3 py-2 rounded-lg ' + (id === teamId ? "bg-slate-800 font-bold" : "text-slate-400") + '">' +
        "<span>" + (i+1) + ". " + esc(t.name) + "</span>" +
        '<span class="tabular-nums">' + valuation(t.ind) + " pts</span></div>"
      ).join("") + "</div>" +
      '<p class="text-center text-slate-500 text-xs mt-4">Esperando a que los moderadores inicien la siguiente ronda…</p>',
      me.name, me.color);
  }

  // final
  const rankingF = Object.entries(game.teams || {}).sort((a,b) => valuation(b[1].ind) - valuation(a[1].ind));
  const myPosF = rankingF.findIndex(([id]) => id === teamId) + 1;
  const medal = myPosF === 1 ? "🏆" : myPosF === 2 ? "🥈" : myPosF === 3 ? "🥉" : "🏦";
  return teamShell(
    '<div class="text-center py-8"><div class="text-6xl mb-3">' + medal + "</div>" +
    '<h3 class="text-3xl font-black">Puesto #' + myPosF + "</h3>" +
    '<p class="text-slate-400 mt-2">' + (myPosF === 1 ? "¡Su neobanco dominó el mercado!" : "Miren la pantalla central para el resumen final.") + "</p>" +
    '<div class="mt-6">' + indicatorsHtml(me.ind) + "</div></div>",
    me.name, me.color);
}
